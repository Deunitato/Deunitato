void reverseArray(int arr[], int size)
{
    int temp , i , j
    i =0;
    j = size - 1;
   while (i<j){
     temp = arr[i]
     arr[i] = arr [ j]
     arr[j] = temp;
     i++;
     j --;

   }

}
